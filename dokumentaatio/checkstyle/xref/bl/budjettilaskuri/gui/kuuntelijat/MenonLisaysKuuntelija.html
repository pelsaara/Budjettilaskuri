11      public MenonLisaysKuuntelija(Rahatilanne rahatilanne, JTextField seliteKentta, JTextField summaKentta, JTextArea tapahtumaKentta, Tulostin tulostin) {
12          super(rahatilanne, seliteKentta, summaKentta, tapahtumaKentta, tulostin);
13      }
14  
15      @Override
16      public void actionPerformed(ActionEvent ae) {
17          int summa = 0;
18          String selite = "";
19          try {
20              summa = Integer.parseInt(summaKentta.getText());
21              selite = seliteKentta.getText();
22          } catch (Exception e) {
23              System.out.println(e.getMessage());
24          }
25  
26          if (summa > 0 && selite.length() <= 26 && selite.length() > 0) {
27              rahatilanne.lisaaMeno(seliteKentta.getText(), summa);
28              seliteKentta.setText("");
29              summaKentta.setText("");
30              tapahtumaKentta.setText(tulostin.tulostaLista(rahatilanne.getMenot()));
31          }
32  
33      }
