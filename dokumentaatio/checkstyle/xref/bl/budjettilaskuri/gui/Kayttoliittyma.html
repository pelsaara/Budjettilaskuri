20  public class Kayttoliittyma implements Runnable {
21  
22      private JFrame frame;
23      private Rahatilanne rahatilanne;
24  
25      public Kayttoliittyma(Rahatilanne rahatilanne) {
26          this.rahatilanne = rahatilanne;
27      }
28  
29      @Override
30      public void run() {
31          frame = new JFrame("Budjettilaskuri");
32          frame.setPreferredSize(new Dimension(600, 400));
33  
34          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
35  
36          luoKomponentit(frame.getContentPane());
37  
38          frame.pack();
39          frame.setVisible(true);
40      }
41  
42      private void luoKomponentit(Container container) {
43          container.add(luoTekstiKentat());
44          container.add(luoBudjettiValikko(), BorderLayout.SOUTH);
45      }
46  
47      private JPanel luoTekstiKentat() {
48          JPanel panel = new JPanel(new GridLayout(3, 2));
49          panel.setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
50  
51          JTextField seliteKentta = new JTextField();
52          JTextField summaKentta = new JTextField();
53  
54          panel.add(new JLabel("Selite: "));
55          panel.add(seliteKentta);
56          panel.add(new JLabel("Summa: "));
57          panel.add(summaKentta);
58  
59          JButton tulo = new JButton("Lis채채 tulo");
60          tulo.addActionListener(new TulonLisaysKuuntelija(rahatilanne, seliteKentta, summaKentta));
61  
62          JButton meno = new JButton("Lis채채 meno");
63          meno.addActionListener(new MenonLisaysKuuntelija(rahatilanne, seliteKentta, summaKentta));
64  
65          panel.add(tulo);
66          panel.add(meno);
67  
68          return panel;
69      }
70  
71      private JPanel luoBudjettiValikko() {
72          JPanel panel = new JPanel(new GridLayout(2, 1));
73          panel.setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
74  
75          JButton budjettiNappi = new JButton("Laske Budjetti");
76          JTextArea tekstiKentta = new JTextArea();
77  
78          budjettiNappi.addActionListener(new BudjetinLaskuKuuntelija(this.rahatilanne, tekstiKentta));
79  
80          panel.add(budjettiNappi);
81          panel.add(tekstiKentta);
82  
83          return panel;
84      }
85  
86      public JFrame getFrame() {
87          return frame;
